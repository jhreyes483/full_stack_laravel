requisitos{
node js, 
} 


actualizar npm 
npm install -g npm@latest

borrar cache npm
npm cache clean --force


desactivar aditorias para evitar fallos
npm set audit false

desinstalar versiones anteriores de angular
npm uninstall -g angular-cli
npm uninstall -g angular/cli
npm cache clean --force

INSTALAR ANGULAR
npm install -g @angular/cli
